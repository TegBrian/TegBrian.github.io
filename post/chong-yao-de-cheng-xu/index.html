一、查找

1、二分查找（时间复杂度为O(nlogn))

详解点这里！

#include<iostream>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<cstdio>
int x,a[10]={1,5,3,6,9,4,7,2,8},s,n;
using namespace std;
int erfen(int a[],int q,int x)
{
	int left=0;
	int right=8;
	int mid=(left+right)/2;
	int k=-1;
	while(left<=right)
	{
		if(a[mid]==x) return mid;
		else if(a[mid]>x) right=mid-1;
		else if(a[mid]<x) left=mid+1;
		mid=(left+right)/2;
	}
	return k;
};
int main()
{
	cin>>n;
	sort(a,a+9);
	int index2 = erfen(a,10,n);
	cout<<index2<< endl;
 	return 0;
}

二、排序

1、lowB三人组（时间复杂度为O(n2n^2n2))

（1）插入排序（时间复杂度为O(n2n^2n2)

详解点这里！

#include<iostream>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<cstdio>
long long n,a[110];
using namespace std;
void insert_sort(long long a[],long long n)
{
	for(int i=1;i<=n;i++)
	{
		long long tmp=a[i];//tmp用来记a[i]的值 
		long long j=i-1;//j指向a[i]所要对比的值 
		while(a[j]>tmp&&a[j]>=1)//1.首先如果a[i]是小于他所要对比的值，a[j]要往右移动 
		{						//2.且如果a[i]比他前面的值都要小，来到了0（此时已越界），就得退出 
			a[j+1]=a[j];
			j--;//a[j]往前移，j递减 
		}
		a[j+1]=tmp;//把a[i]归位 
	}
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++) cin>>a[i];
	insert_sort(a,n);
	for(int i=1;i<=n;i++) cout<<a[i]<<" ";


 	return 0;
}

（2）冒泡排序（时间复杂度为O(n2n^2n2))

详解点这里！

//优化版 
//时间复杂度=O(n2) 
#include<iostream>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<cstdio> 
long long n,a[110]; 
using namespace std;
void bubble_sort(long long a[],long long n)
{
	bool ok=false;//拿ok来判是否交换 
	for(int i=1;i<=n-1;i++) //由于将n-1个数已经排好，剩下的一个已经不用排了 
	{
		ok=false;
		for(int j=1;j<=n-1;j++) //从第一个数开始比较 (这里要n-1因为最后一个会和0比，他一定大于0，会被排出去） 
			if(a[j]>a[j+1])
			{
				swap(a[j],a[j+1]);//把小的放在前头 
				ok=true;//有交换判true
			}	
		if(!ok)//如果数组并没有交换
		{
			return;
		} 
	}	
	
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++) cin>>a[i];
	bubble_sort(a,n);
	for(int i=1;i<=n;i++) cout<<a[i]<<" ";
 	return 0;
}

（3）选择排序（时间复杂度为O(n2n^2n2))

详解点这里！

#include<iostream>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<cstdio>
long long n,a[110];
using namespace std;
void select_sort(long long a[],long long n)
{
	long long mi=100000,s=1,k=0;//用s来记递增的下标，k来记最小值的下标 
	for(int i=1;i<=n-1;i++)//由于将n-1个数已经排好，剩下的一个已经不用排了
	{
		mi=100000;
		for(int j=s;j<=n;j++) 
		{
			if(mi>a[j]) 
			{
			mi=a[j]; 	
			k=j;
			}
		}
		swap(a[s],a[k]);//互换 
		s++;//s递增 
	}
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	cin>>a[i];
	select_sort(a,n);
	for(int i=1;i<=n;i++) cout<<a[i]<<" ";

 	return 0;
}

2、NB三人组（时间复杂度为O(nlogn))

（1）堆排序

详解点这里！

 #include<iostream>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<cstdio>
int a[110],n;
using namespace std;
void kkk(int a[],int low,int high)
{
	//low:堆的根结点位置
	//high:堆的最后一个元素的位置 
	int x=low;//x开始指向根节点 
	int y=2*x+1;//y开始是左孩子 
	int tmp=a[low];//把堆顶存起来 
	while(y<=high)//只要y位置有值
	{
		if(a[y+1]>a[y]&&y+1<=high)//如果右孩子比左边大且有值 
		{
			y=y+1;//y指向右孩子 
		}
		if(a[y]>tmp) 
		{
			a[x]=a[y];	  
			x=y;//往下看一层 
			y=2*x+1;
		}
		else//tmp更大，把tmp放到x的位置 
		{
			a[x]=tmp; 
			break;
		}
	} 
	a[x]=tmp; //把tmp放到叶子节点上 
}
void dui_pai_xu(int a[],int n)
{
	for(int i=(n-1)/2;i>=0;i--)//i表示建堆的时候调整的部分的下标 
	{
		kkk(a,i,n);
	}
	for(int i=n-1;i>0;i--)//i指向当前堆最后一个元素 
	{
		swap(a[0],a[i]);
		kkk(a,0,i-1);
	}
} 
int main()
{
	cin>>n;
	for(int i=0;i<n;i++) cin>>a[i];
	dui_pai_xu(a,n);
	for(int i=0;i<n;i++) cout<<a[i]<<" ";
 	return 0;
}
